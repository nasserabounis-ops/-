<script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
    import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
    import { getFirestore, collection, doc, setDoc, onSnapshot, getDocs, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

    let db, auth, currentUser, appId;

    window.submitCloudPoll = async function(e, formEl) {
        e.preventDefault();
        
        if (!currentUser) {
            document.getElementById('poll-error-msg').innerText = "لم يتم الاتصال بالسحابة بعد، يرجى الانتظار.";
            document.getElementById('poll-error-msg').classList.remove('hidden');
            return;
        }

        const btn = document.getElementById('submit-poll-btn');
        btn.disabled = true;
        btn.innerHTML = `<span class="animate-spin">⏳</span> جاري الإرسال...`;

        const bgVal = document.getElementById('poll-background').value;
        const chalVal = document.getElementById('poll-challenge').value;

        try {
            const pollDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'audience_polls', currentUser.uid);
            await setDoc(pollDocRef, { 
                background: bgVal, 
                challenge: chalVal,
                timestamp: Date.now() 
            });

            btn.innerHTML = "تم تسجيل رأيك سحابياً ✓";
            document.getElementById('poll-error-msg').classList.add('hidden');
            const msg = document.getElementById('poll-success-msg');
            msg.classList.remove('hidden');
            msg.classList.add('animate-pulse');
            setTimeout(() => msg.classList.remove('animate-pulse'), 2000);
            
            setTimeout(() => {
                btn.disabled = false;
                btn.innerHTML = "تحديث رأيي 🔄";
                msg.classList.add('hidden');
            }, 3000);

        } catch (error) {
            console.error("Error saving poll:", error);
            btn.disabled = false;
            btn.innerHTML = "إرسال رأيي للسحابة 🚀";
            document.getElementById('poll-error-msg').innerText = "حدث خطأ أثناء الإرسال.";
            document.getElementById('poll-error-msg').classList.remove('hidden');
        }
    };

    window.confirmReset = async function() {
        const pwdInput = document.getElementById('admin-password');
        const pwdError = document.getElementById('pwd-error');
        const loader = document.getElementById('reset-loader');

        // Secret Password Validation using simple Base64 to hide 'admin2026' from plain sight
        if(btoa(pwdInput.value) === 'YWRtaW4yMDI2') { 
            if (!currentUser) return;
            
            pwdError.classList.add('hidden');
            loader.classList.remove('hidden');

            try {
                const pollsRef = collection(db, 'artifacts', appId, 'public', 'data', 'audience_polls');
                const snapshot = await getDocs(pollsRef);
                
                const deletePromises = [];
                snapshot.forEach((docSnap) => {
                    deletePromises.push(deleteDoc(docSnap.ref));
                });
                
                await Promise.all(deletePromises);
                
                window.closePwdModal();
                const btn = document.getElementById('submit-poll-btn');
                btn.disabled = false;
                btn.innerHTML = "إرسال رأيي للسحابة 🚀";
                document.getElementById('poll-success-msg').classList.add('hidden');
                document.getElementById('audiencePollForm').reset();
            } catch (error) {
                console.error("Error resetting data:", error);
                alert("حدث خطأ أثناء عملية المسح.");
            } finally {
                loader.classList.add('hidden');
            }
        } else {
            pwdError.classList.remove('hidden');
            pwdInput.value = '';
            pwdInput.focus();
        }
    };

    async function initFirebase() {
        try {
            const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
            if(!firebaseConfig) {
                console.log("No Firebase config found. Local mode.");
                enableGateway();
                return;
            }

            const app = initializeApp(firebaseConfig);
            auth = getAuth(app);
            db = getFirestore(app);
            appId = typeof __app_id !== 'undefined' ? __app_id : 'default-startup-app';

            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                await signInWithCustomToken(auth, __initial_auth_token);
            } else {
                await signInAnonymously(auth);
            }

            onAuthStateChanged(auth, (user) => {
                currentUser = user;
                if (user) {
                    enableGateway();
                    startLivePollListener();
                }
            });

        } catch(e) { 
            console.error("Firebase init error:", e); 
            enableGateway();
        }
    }

    function enableGateway() {
        document.getElementById('cloud-sync-status').innerText = "متصل بالسحابة وجاهز للمزامنة! ✅";
        document.getElementById('cloud-sync-status').classList.remove('animate-pulse');
        document.getElementById('cloud-sync-status').classList.replace('bg-blue-800/50', 'bg-emerald-800/50');
        document.getElementById('cloud-sync-status').classList.replace('border-blue-500/30', 'border-emerald-500/30');
        document.getElementById('cloud-sync-status').classList.replace('text-blue-200', 'text-emerald-200');
        
        document.getElementById('btn-enter-fs').disabled = false;
        document.getElementById('btn-enter-normal').disabled = false;
    }

    function startLivePollListener() {
        const pollsRef = collection(db, 'artifacts', appId, 'public', 'data', 'audience_polls');
        onSnapshot(pollsRef, (snapshot) => {
            let aggregatedData = {
                backgrounds: { academic: 0, practical: 0, tech: 0 },
                challenges: { funding: 0, pmf: 0, team: 0, market: 0 }
            };

            snapshot.forEach((docSnap) => {
                const data = docSnap.data();
                if (data.background && aggregatedData.backgrounds[data.background] !== undefined) {
                    aggregatedData.backgrounds[data.background]++;
                }
                if (data.challenge && aggregatedData.challenges[data.challenge] !== undefined) {
                    aggregatedData.challenges[data.challenge]++;
                }
            });

            if (window.updatePollChartsUI) {
                window.updatePollChartsUI(aggregatedData);
            }
        }, (error) => {
            console.error("Firestore sync error:", error);
        });
    }

    document.getElementById('audiencePollForm').addEventListener('submit', function(e) {
        window.submitCloudPoll(e, this);
    });

    initFirebase();
</script>

        else if (docElm.webkitRequestFullScreen) { p = docElm.webkitRequestFullScreen(); } 
        else if (docElm.msRequestFullscreen) { p = docElm.msRequestFullscreen(); }
        
        // Handle blocked iframe scenarios using Immersive Mode
        if(p && typeof p.catch === 'function') {
            p.catch(err => {
                console.warn("Native Fullscreen blocked by browser/iframe policy. Activating Immersive Mode fallback.");
                document.body.classList.add('pseudo-fullscreen');
                updateFullscreenUI(true);
            });

    window.updatePollChartsUI = function(liveData) {
        const totalBgs = Object.values(liveData.backgrounds).reduce((a, b) => a + b, 0);
        const totalChals = Object.values(liveData.challenges).reduce((a, b) => a + b, 0);
        
        // Backgrounds Chart Update
        if(totalBgs === 0) {
            document.getElementById('empty-state-bg').classList.remove('opacity-0', 'pointer-events-none');
        } else {
            document.getElementById('empty-state-bg').classList.add('opacity-0', 'pointer-events-none');
            pollBgChart.data.datasets[0].data = [liveData.backgrounds.academic, liveData.backgrounds.practical, liveData.backgrounds.tech];
            pollBgChart.update();
        }

        // Challenges Chart Update
        if(totalChals === 0) {
            document.getElementById('empty-state-chal').classList.remove('opacity-0', 'pointer-events-none');
        } else {
            document.getElementById('empty-state-chal').classList.add('opacity-0', 'pointer-events-none');
            pollChalChart.data.datasets[0].data = [liveData.challenges.funding, liveData.challenges.pmf, liveData.challenges.team, liveData.challenges.market];
            pollChalChart.update();
        }
    };

            pwdModalContent.classList.remove('scale-95');
        }, 10);
        pwdInput.focus();
    };

        const pwdModalContent = document.getElementById('password-modal-content');
        pwdModal.classList.add('opacity-0');
        pwdModalContent.classList.add('scale-95');
        setTimeout(() => {
            pwdModal.classList.add('hidden');
        }, 300);
    };
